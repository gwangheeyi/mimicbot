현재 단계는 이렇게 진행된 것

Gazebo 카메라
   ↓ 정상
/gazebo/camera/image
   ↓ ros_gz_bridge
/camera/image_raw
   ↓ web_video_server
HTTP 영상
   ↓
Flutter WebView

지금 출력은 카메라 생성에 성공했다는 뜻이며, 다음 핵심 명령은 이것입니다.

ros2 run ros_gz_bridge parameter_bridge \
'/gazebo/camera/image@sensor_msgs/msg/Image[gz.msgs.Image' \
--ros-args \
-r /gazebo/camera/image:=/camera/image_raw

camera_test.sdf 에서
빨간 상자를 움직이게 하려면
<model name="red_box">
  <static>true</static>

카메라를 움직이게 하려면
<model name="fixed_camera">
  <static>false</static>

카메라를 떨어 뜨리지 않게 하려면
<model name="fixed_camera">
  <static>false</static>

  <link name="camera_link">
    <gravity>false</gravity>

카메라를 아애롸 같이 수정
<model name="fixed_camera">
  <static>false</static>

  <pose>3 0 1.5 0 0 3.141592</pose>

  <link name="camera_link">

    <!-- 카메라 모델이 중력으로 떨어지지 않게 합니다. -->
    <gravity>false</gravity>

    <visual name="camera_body">
      <geometry>
        <box>
          <size>0.2 0.15 0.15</size>
        </box>
      </geometry>

      <material>
        <ambient>0.1 0.1 0.1 1</ambient>
        <diffuse>0.1 0.1 0.1 1</diffuse>
      </material>
    </visual>

    <sensor name="rgb_camera" type="camera">
      <always_on>true</always_on>
      <update_rate>30</update_rate>
      <topic>/gazebo/camera/image</topic>
      <visualize>true</visualize>

      <camera>
        <horizontal_fov>1.047</horizontal_fov>

        <image>
          <width>640</width>
          <height>480</height>
          <format>R8G8B8</format>
        </image>

        <clip>
          <near>0.1</near>
          <far>100</far>
        </clip>
      </camera>
    </sensor>

  </link>
</model>
